# requirements.txt — Python dependencies

# pip installs these inside the Docker image during the build step.

# gunicorn — production-grade WSGI server.
# Flask's built-in server (app.run) is single-threaded and not suitable
# for production. gunicorn spawns multiple worker processes to handle
# concurrent requests efficiently.
gunicorn==23.0.0
